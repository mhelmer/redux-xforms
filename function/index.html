    import createByKey from 'redux-xforms/createByKey.js'

    Creates a reducer transformer that adds byKey-filtering.

    Params:

    NameTypeAttributeDescription
    predicate function(action: Object): bool

    The predicate to use
    mapActionToKey function(action: Object): string

    Map action to the key of the selected slice.

    Return:

    function

    A reducer transformer

    Example:

    const byKeyReducer = createByKey(
  action => action.hasOwnProperty('userId'),
  action => action.userId
)(reducer)

// returns {
//   2: { username: 'mrXform' },
//   3: reducer({ username: 'otherUser' }, action),
// }
byKeyReducer({
  { 2: { username: 'mrXform' } },
  { 3: { username: 'otherUser' } },
}, { userId: 3 })

    public isolateSlice(mapActionToSlice: function(action: Object): string): function source 

    import isolateSlice from 'redux-xforms/isolateSlice.js'

    Creates a reducer transformer that calls the reducer with only a slice of the state, based on the action.

    Make sure to put a filtering xform in front of this, so actions that can't be mapped wil not run. It also expects state to be initialized as an Object.

    Params:

    NameTypeAttributeDescription
    mapActionToSlice function(action: Object): string

    Map action to the key of the updating slice.

    Return:

    function

    A reducer transformer

    Example:

    const toIdSlice = isolateSlice(action => action.id)(reducer)
// returns reducer({ username: 'xForman' }, { id: 2 })
toIdSlice(
  { 2: { username: 'xForman' } },
  { id: 2 }
)

    public transformReducers(transformers: Object): function source 

    import transformReducers from 'redux-xforms/transformReducers.js'

    Transforms an object with reducers into an object with the original reducers merged with transformed reducers for matching keys. It can for example be composed with combineReducers or createReducersByKey.

    Params:

    NameTypeAttributeDescription
    transformers Object

    The reducer transformers

    Return:

    function

    The reducer transformer

    Example:

    const reducer = (state = null, action) => {
  return action.type === 'SUCCESS' ? action.payload
    : state
}
const enhance = reducer => (state = {}, action) => ({
  ...state,
  enhanced: reducer(state.enhanced, action),
})
const reducers = {
  a: reducer,
  b: reducer,
}
const transformedReducers = transformReducers({
  a: enhance,
})(reducers)

// returns { enhanced: 'some-payload' }
[{}, {
  type: 'SUCCESS',
  payload: 'some-payload',
}].reduce(transformedReducers.a, undefined)

// returns null
transformedReducers.b(undefined, {})

    import updateSlice from 'redux-xforms/updateSlice.js'

    Creates a reducer transformer that updates only one slice of the state based on the action.

    Make sure to put a filtering xform in front of this, so actions that can't be mapped wil not run. It also expects state to be initialized as an Object.

    Params:

    NameTypeAttributeDescription
    mapActionToSlice function(action: Object): string

    Map action to the key of the updating slice.

    Return:

    function

    A reducer transformer

    Example:

    const reduceSlice = updateSlice(action => action.id)(reducer)
// returns {
//   2: reducer({ username: 'xForman' }, action),
// }
reduceSlice(
  { 2: { username: 'xForman' }, 3: { username: 'yUser' } },
  { id: 2 }
)
